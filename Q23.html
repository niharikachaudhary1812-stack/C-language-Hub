// Check numbers from user is palindrome or not ?
#include
#include
void main ()
{
int num,rev=0,rem,temp;
printf("Enter a number :");
scanf("%d",&num);
temp=num;
while(num>0)
{
rem=num%10;
rev=rev*10+rem;
num=num/10;
}
if(temp==rev)
printf("The number is a palindrome.");
else
printf("The number is not a palindrome.");
}
/*
Output :
Enter a number :52
The number is not a palindrome.
*/
